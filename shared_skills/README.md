cp -R ~/Development/vibe_coding/shared_skills/* ~/.codex/skills/

ln -s ~/Development/vibe_coding/shared_skills/rm-work-code ~/.codex/skills/
ln -s ~/Development/vibe_coding/shared_skills/rm-doc-rate ~/.codex/skills/
